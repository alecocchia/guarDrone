# === ACADOS Installation === 
USER root
WORKDIR /opt
RUN git clone --recursive https://github.com/acados/acados.git

WORKDIR /opt/acados
RUN mkdir -p build && cd build && \
    cmake .. \
      -DCMAKE_INSTALL_PREFIX=/opt/acados \
      -DACADOS_WITH_QPOASES=ON \
      -DACADOS_WITH_OPENMP=ON \
      -DBUILD_SHARED_LIBS=ON \
      -DACADOS_INSTALL_DIR=/opt/acados \
      -DBLASFEO_TARGET=GENERIC \
      -DHPIPM_TARGET=GENERIC && \
    make -j$(nproc) && make install

ENV ACADOS_SOURCE_DIR="/opt/acados"
ENV ACADOS_INSTALL_DIR="/opt/acados"
ENV LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/opt/acados/lib
ENV OMP_NUM_THREADS=4

RUN pip3 install --upgrade --break-system-packages pip setuptools wheel \
    && pip3 install --break-system-packages /opt/acados/interfaces/acados_template

RUN mkdir -p /opt/acados/bin && \
    curl -L https://github.com/acados/tera_renderer/releases/download/v0.2.0/t_renderer-v0.2.0-linux-amd64 \
    -o /opt/acados/bin/t_renderer && \
    chmod +x /opt/acados/bin/t_renderer
   
# Add acados library paths to bashrc
RUN echo "export LD_LIBRARY_PATH=\$LD_LIBRARY_PATH:/opt/acados/lib" >> /root/.bashrc

# Entrypoint automatico (con link a tmux.conf)
RUN echo '#!/bin/bash\n\

[ -f /root/my_ros2_ws/SimulationScripts/tmux.conf ] && ln -sf /root/my_ros2_ws/SimulationScripts/tmux.conf /root/.tmux.conf\n\
exec "$@"' > /entrypoint.sh && chmod +x /entrypoint.sh

ENTRYPOINT ["/entrypoint.sh"]
CMD ["bash"]
