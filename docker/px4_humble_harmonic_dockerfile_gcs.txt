FROM px4io/px4-dev-base-jammy:latest

ENV DEBIAN_FRONTEND=noninteractive \
    LANG=en_US.UTF-8 \
    LC_ALL=en_US.UTF-8

WORKDIR /root

# Update and install tools
RUN apt-get update && apt-get install -y --no-install-recommends \
    gedit \
    git \
    bash-completion \
    locales \
    software-properties-common \
    curl \
    python3-pip \
    wget \
    htop \
    iputils-ping \
    net-tools \
    iproute2 \
    tree \
    unzip \
    vim \
    nano \
    && locale-gen en_US en_US.UTF-8 \
    && update-locale LC_ALL=en_US.UTF-8 LANG=en_US.UTF-8 \
    && add-apt-repository -y universe \
    && rm -rf /var/lib/apt/lists/*

# ROS2 Humble
RUN curl -sSL https://raw.githubusercontent.com/ros/rosdistro/master/ros.key \
      -o /usr/share/keyrings/ros-archive-keyring.gpg \
    && echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/ros-archive-keyring.gpg] \
        http://packages.ros.org/ros2/ubuntu $(. /etc/os-release && echo $UBUNTU_CODENAME) main" \
        > /etc/apt/sources.list.d/ros2.list \
    && apt-get update \
    && apt-get install -y ros-humble-desktop ros-dev-tools \
    python3-rosdep \
    python3-vcstool \
    python3-colcon-common-extensions \
    && echo "source /opt/ros/humble/setup.bash" >> /root/.bashrc \
    && python3 -m pip install --user -U empy==3.3.4 pyros-genmsg setuptools \
    && rm -rf /var/lib/apt/lists/*


# C'erano Gazebo e bridge

RUN wget https://raw.githubusercontent.com/mavlink/mavros/master/mavros/scripts/install_geographiclib_datasets.sh \
    && bash ./install_geographiclib_datasets.sh   

RUN apt-get update && apt-get install -y --no-install-recommends \     
    libcanberra-gtk-module libcanberra-gtk3-module \
    && rm -rf /var/lib/apt/lists/*

# Micro-XRCE-DDS-Agent
RUN git clone https://github.com/eProsima/Micro-XRCE-DDS-Agent.git \
    && cd Micro-XRCE-DDS-Agent \
    && mkdir build \
	&& cd build \
	&& cmake .. \
	&& make \
	&& make install \
	&& ldconfig /usr/local/lib/

# ROS2 px4 workspace
RUN mkdir -p /root/px4_ws/src \
    && cd ~/px4_ws/src \
    && git clone --single-branch -b release/1.16 https://github.com/PX4/px4_msgs px4_msgs \
    && git clone --single-branch -b release/1.16 https://github.com/PX4/px4_ros_com.git \
    && cd .. \
    && . /opt/ros/humble/setup.sh \
    && colcon build --parallel-workers $(nproc) --symlink-install --cmake-args -DCMAKE_BUILD_TYPE=Release
    
RUN echo "source /root/px4_ws/install/local_setup.bash" >> /root/.bashrc

# Additional packages
RUN apt-get update && apt-get install -y --no-install-recommends \    
    aptitude \
    tmux \
    tmuxp \
    && rm -rf /var/lib/apt/lists/*

RUN  git config --global --add safe.directory '*'
USER root

# c'era la parte relativa alla camera

# CycloneDDS (per comunicazione multi-drone)
RUN apt-get update && apt-get install -y --no-install-recommends \ 
    ros-humble-rmw-cyclonedds-cpp \
    && rm -rf /var/lib/apt/lists/*

ENV RMW_IMPLEMENTATION=rmw_cyclonedds_cpp

# Python deps per gcs_pkg (supervisor + logger)
RUN pip3 install --break-system-packages numpy scipy

# ros2_control (necessario per i pacchetti haptic Force Dimension)
RUN apt-get update && apt-get install -y --no-install-recommends \  
    libccd-dev \
    libboost-thread-dev \
    libboost-system-dev \
    libboost-filesystem-dev \
    libboost-program-options-dev \
    libeigen3-dev \
    libompl-dev \
    ros-humble-sophus* \
    && rm -rf /var/lib/apt/lists/*

RUN git clone --single-branch -b fcl-0.5 https://github.com/flexible-collision-library/fcl.git \
    && cd fcl \
    && mkdir build \
    && cd build \
    && cmake .. \
    && make -j$(nproc) \
    && make install

RUN  apt-get update && apt-get install -y --no-install-recommends \  
     ros-humble-navigation2 \
     ros-humble-nav2-bringup \
     ros-humble-tf-transformations \
     ros-humble-robot-localization \
     ros-humble-ros2-control \
     ros-humble-ros2-controllers \
     python-is-python3 \
     && rm -rf /var/lib/apt/lists/*

# Install ARM toolchain for PX4 hardware compilation
RUN wget https://armkeil.blob.core.windows.net/developer/Files/downloads/gnu-rm/10.3-2021.10/gcc-arm-none-eabi-10.3-2021.10-x86_64-linux.tar.bz2 \
    && tar -xjf gcc-arm-none-eabi-10.3-2021.10-x86_64-linux.tar.bz2 -C /opt \
    && rm gcc-arm-none-eabi-10.3-2021.10-x86_64-linux.tar.bz2

ENV PATH="/opt/gcc-arm-none-eabi-10.3-2021.10/bin:${PATH}"

# ros2_unbag (Corretto per funzionare senza sudo)
RUN apt-get update \
    && apt-get install -y --no-install-recommends libxcb-cursor0 libxcb-shape0 libxcb-icccm4 libxcb-keysyms1 libxkbcommon-x11-0 \
    && rm -rf /var/lib/apt/lists/* \
    && pip3 install --break-system-packages ros2-unbag

# === GCS ROS2 packages (sparse clone dal monorepo) ===
RUN git clone --depth 1 --filter=blob:none --sparse \
        -b master \
        https://github.com/alecocchia/guarDrone.git /tmp/guardrone \
    && cd /tmp/guardrone \
    && git sparse-checkout set \
        my_ros2_ws/src/utils_pkg \
        my_ros2_ws/src/gcs_pkg \
        my_ros2_ws/src/haptic_device \
    && mkdir -p /root/my_ros2_ws/src \
    && cp -r /tmp/guardrone/my_ros2_ws/src/utils_pkg /root/my_ros2_ws/src/ \
    && cp -r /tmp/guardrone/my_ros2_ws/src/gcs_pkg /root/my_ros2_ws/src/ \
    && cp -r /tmp/guardrone/my_ros2_ws/src/haptic_device /root/my_ros2_ws/src/ \
    && rm -rf /tmp/guardrone

# Entrypoint automatico (con link a tmux.conf)
RUN echo '#!/bin/bash\n\
set -e\n\
source /opt/ros/humble/setup.bash\n\
[ -f /root/px4_ws/install/setup.bash ] && source /root/px4_ws/install/setup.bash\n\
[ -f /root/my_ros2_ws/install/setup.bash ] && source /root/my_ros2_ws/install/setup.bash\n\
# Crea il symlink per tmux se il file esiste nel volume montato\n\
[ -f /root/my_ros2_ws/SimulationScripts/tmux.conf ] && ln -sf /root/my_ros2_ws/SimulationScripts/tmux.conf /root/.tmux.conf\n\
exec "$@"' > /entrypoint.sh && chmod +x /entrypoint.sh

ENTRYPOINT ["/entrypoint.sh"]
CMD ["bash"]
